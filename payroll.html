// ══════════════════════════════════════════════════════════ // PAYROLL STAFF DIRECTORY + EXCLUSIONS + PAYMENT HISTORY // ══════════════════════════════════════════════════════════ // // TABS (auto-created if missing): // "Directory" — Name | Sort Code | Account Number | Entity // "Excluded" — Name | Sort Code | Account Number | Entity // "History" — Date | Month | Entity | Name | Sort Code | Account Number | Gross | Deductions | Deduction Details | Net Pay | Reference // // After pasting: Deploy → Manage deployments → edit → New version → Deploy function getSheet(tabName, headers) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(tabName); if (!sheet) { sheet = ss.insertSheet(tabName); if (headers && headers.length) { sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold'); sheet.setFrozenRows(1); } } return sheet; } function getDirectorySheet() { return getSheet('Directory', ['Name','Sort Code','Account Number','Entity']); } function getExcludedSheet() { return getSheet('Excluded', ['Name','Sort Code','Account Number','Entity']); } function getHistorySheet() { return getSheet('History', ['Date','Month','Entity','Name','Sort Code','Account Number','Gross','Deductions','Deduction Details','Net Pay','Reference']); } // ── Routing ── function doGet(e) { var action = (e && e.parameter && e.parameter.action) || 'list'; try { if (action === 'list') return listStaff(); if (action === 'excluded') return listExcluded(); if (action === 'history') return getHistory(e.parameter); return jsonResponse({ error: 'Unknown action' }); } catch (err) { return jsonResponse({ error: err.toString() }); } } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var a = data.action; if (a === 'add') return addStaff(data); if (a === 'remove') return removeStaff(data); if (a === 'addBatch') return addBatch(data); if (a === 'exclude') return excludeStaff(data); if (a === 'unexclude') return unexcludeStaff(data); if (a === 'logPayments') return logPayments(data); return jsonResponse({ error: 'Unknown action' }); } catch (err) { return jsonResponse({ error: err.toString() }); } } // ══════════════════════════════════════════════ // DIRECTORY // ══════════════════════════════════════════════ function readSheet4Col(sheet) { var lastRow = sheet.getLastRow(); if (lastRow < 2) return []; var data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); var result = []; for (var i = 0; i < data.length; i++) { if (!data[i][0] || !data[i][1] || !data[i][2]) continue; result.push({ name: String(data[i][0]).trim(), sortCode: String(data[i][1]).trim(), accountNumber: String(data[i][2]).trim(), entity: String(data[i][3] || 'both').trim().toLowerCase(), rowIndex: i + 2 }); } return result; } function listStaff() { return jsonResponse({ staff: readSheet4Col(getDirectorySheet()) }); } function listExcluded() { return jsonResponse({ excluded: readSheet4Col(getExcludedSheet()) }); } function addStaff(data) { if (!data.name || !data.sortCode || !data.accountNumber) return jsonResponse({ error: 'Missing fields' }); var sheet = getDirectorySheet(); if (isDuplicate(sheet, data.sortCode, data.accountNumber)) return jsonResponse({ error: 'Already exists', duplicate: true }); sheet.appendRow([data.name, data.sortCode, data.accountNumber, data.entity || 'both']); sheet.getRange(sheet.getLastRow(), 2, 1, 2).setNumberFormat('@'); return jsonResponse({ success: true }); } function addBatch(data) { if (!data.staff || !data.staff.length) return jsonResponse({ error: 'No staff' }); var dirSheet = getDirectorySheet(); var exSheet = getExcludedSheet(); var dirKeys = getKeys(dirSheet); var exKeys = getKeys(exSheet); var newRows = []; for (var j = 0; j < data.staff.length; j++) { var s = data.staff[j]; if (!s.name || !s.sortCode || !s.accountNumber) continue; var key = s.sortCode + '|' + s.accountNumber; if (dirKeys[key] || exKeys[key]) continue; // skip if already in directory OR excluded dirKeys[key] = true; newRows.push([s.name, s.sortCode, s.accountNumber, s.entity || 'both']); } if (newRows.length) { var lastRow = dirSheet.getLastRow(); dirSheet.getRange(lastRow + 1, 1, newRows.length, 4).setValues(newRows); dirSheet.getRange(lastRow + 1, 2, newRows.length, 2).setNumberFormat('@'); } return jsonResponse({ success: true, added: newRows.length }); } function removeStaff(data) { return removeFromSheet(getDirectorySheet(), data.sortCode, data.accountNumber); } // ══════════════════════════════════════════════ // EXCLUSIONS // ══════════════════════════════════════════════ function excludeStaff(data) { if (!data.name || !data.sortCode || !data.accountNumber) return jsonResponse({ error: 'Missing fields' }); var exSheet = getExcludedSheet(); if (isDuplicate(exSheet, data.sortCode, data.accountNumber)) return jsonResponse({ error: 'Already excluded', duplicate: true }); // Add to Excluded exSheet.appendRow([data.name, data.sortCode, data.accountNumber, data.entity || 'both']); exSheet.getRange(exSheet.getLastRow(), 2, 1, 2).setNumberFormat('@'); // Remove from Directory if present removeFromSheetSilent(getDirectorySheet(), data.sortCode, data.accountNumber); return jsonResponse({ success: true, message: 'Excluded ' + data.name }); } function unexcludeStaff(data) { if (!data.sortCode || !data.accountNumber) return jsonResponse({ error: 'Missing fields' }); // Find details before removing var exSheet = getExcludedSheet(); var lastRow = exSheet.getLastRow(); var name = data.name || '', entity = data.entity || 'both'; if (lastRow >= 2) { var rows = exSheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (var i = 0; i < rows.length; i++) { if (String(rows[i][1]).trim() === data.sortCode && String(rows[i][2]).trim() === data.accountNumber) { name = String(rows[i][0]).trim(); entity = String(rows[i][3] || 'both').trim(); break; } } } // Remove from Excluded removeFromSheetSilent(exSheet, data.sortCode, data.accountNumber); // Add back to Directory var dirSheet = getDirectorySheet(); if (!isDuplicate(dirSheet, data.sortCode, data.accountNumber)) { dirSheet.appendRow([name, data.sortCode, data.accountNumber, entity]); dirSheet.getRange(dirSheet.getLastRow(), 2, 1, 2).setNumberFormat('@'); } return jsonResponse({ success: true, message: 'Restored ' + name }); } // ══════════════════════════════════════════════ // HISTORY // ══════════════════════════════════════════════ function logPayments(data) { if (!data.payments || !data.payments.length) return jsonResponse({ error: 'No payments' }); var sheet = getHistorySheet(); var now = new Date(); var dateStr = Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm'); var monthStr = data.month || Utilities.formatDate(now, Session.getScriptTimeZone(), 'MMM yyyy'); var entity = data.entity || 'unknown'; var ref = data.reference || 'Payroll'; var rows = [], totalNet = 0; for (var i = 0; i < data.payments.length; i++) { var p = data.payments[i]; var gross = p.grossAmount || p.amount || 0; var deds = p.deductions || 0; var dedDetails = p.deductionDetails || ''; var net = p.amount || (gross - deds); rows.push([dateStr, monthStr, entity, p.name || '', p.sortCode || '', p.accountNumber || '', gross, deds, dedDetails, net, ref]); totalNet += net; } if (rows.length) { var lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1, rows.length, 11).setValues(rows); sheet.getRange(lastRow + 1, 5, rows.length, 2).setNumberFormat('@'); sheet.getRange(lastRow + 1, 7, rows.length, 1).setNumberFormat('£#,##0.00'); sheet.getRange(lastRow + 1, 8, rows.length, 1).setNumberFormat('£#,##0.00'); sheet.getRange(lastRow + 1, 10, rows.length, 1).setNumberFormat('£#,##0.00'); } return jsonResponse({ success: true, logged: rows.length, total: totalNet }); } function getHistory(params) { var sheet = getHistorySheet(), lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ history: [] }); var data = sheet.getRange(2, 1, lastRow - 1, 11).getValues(), history = []; var fEntity = params && params.entity ? params.entity.toLowerCase() : null; var fMonth = params && params.month ? params.month : null; for (var i = 0; i < data.length; i++) { var e = { date: String(data[i][0]), month: String(data[i][1]), entity: String(data[i][2]), name: String(data[i][3]), sortCode: String(data[i][4]), accountNumber: String(data[i][5]), gross: Number(data[i][6]), deductions: Number(data[i][7]), deductionDetails: String(data[i][8]), netPay: Number(data[i][9]), reference: String(data[i][10]) }; if (fEntity && e.entity.toLowerCase() !== fEntity) continue; if (fMonth && e.month !== fMonth) continue; history.push(e); } return jsonResponse({ history: history }); } // ══════════════════════════════════════════════ // HELPERS // ══════════════════════════════════════════════ function getKeys(sheet) { var lastRow = sheet.getLastRow(), keys = {}; if (lastRow < 2) return keys; var data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (var i = 0; i < data.length; i++) keys[String(data[i][1]).trim() + '|' + String(data[i][2]).trim()] = true; return keys; } function isDuplicate(sheet, sortCode, accountNumber) { var lastRow = sheet.getLastRow(); if (lastRow < 2) return false; var data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (var i = 0; i < data.length; i++) { if (String(data[i][1]).trim() === sortCode && String(data[i][2]).trim() === accountNumber) return true; } return false; } function removeFromSheet(sheet, sortCode, accountNumber) { if (!sortCode || !accountNumber) return jsonResponse({ error: 'Missing fields' }); var lastRow = sheet.getLastRow(); if (lastRow < 2) return jsonResponse({ error: 'Empty' }); var data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (var i = data.length - 1; i >= 0; i--) { if (String(data[i][1]).trim() === sortCode && String(data[i][2]).trim() === accountNumber) { sheet.deleteRow(i + 2); return jsonResponse({ success: true }); } } return jsonResponse({ error: 'Not found' }); } function removeFromSheetSilent(sheet, sortCode, accountNumber) { var lastRow = sheet.getLastRow(); if (lastRow < 2) return; var data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); for (var i = data.length - 1; i >= 0; i--) { if (String(data[i][1]).trim() === sortCode && String(data[i][2]).trim() === accountNumber) { sheet.deleteRow(i + 2); return; } } } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
